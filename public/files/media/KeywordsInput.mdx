# KeywordsInput

A form input component for managing keywords and tags. Users can add keywords by typing and pressing Enter or comma, and remove them by clicking the X button or using backspace.

### **Import**
```tsx
import { KeywordsInput } from '@app-studio/web';
```

### **Basic Usage**
A simple keywords input with default settings.

```tsx
import React, { useState } from 'react';
import { KeywordsInput } from '../KeywordsInput';

export const BasicKeywordsInput = () => {
  const [keywords, setKeywords] = useState(['react', 'typescript']);

  return (
    <KeywordsInput
      label="Skills"
      value={keywords}
      onChange={setKeywords}
      placeholder="Type and press Enter to add keywords..."
      helperText="Add relevant skills for your profile"
    />
  );
};
```

### **Custom Separators**
Configure which keys trigger keyword creation.

```tsx
import React, { useState } from 'react';
import { KeywordsInput } from '../KeywordsInput';

export const CustomSeparatorsExample = () => {
  const [keywords, setKeywords] = useState([]);

  return (
    <KeywordsInput
      label="Custom Separators"
      value={keywords}
      onChange={setKeywords}
      separators={['enter', 'comma', 'space', 'tab']}
      placeholder="Use Enter, comma, space, or tab to add keywords..."
      helperText="Multiple separator options available"
    />
  );
};
```

### **Validation and Limits**
Control keyword validation and limits.

```tsx
import React, { useState } from 'react';
import { KeywordsInput } from '../KeywordsInput';

export const ValidationExample = () => {
  const [keywords, setKeywords] = useState(['react']);

  return (
    <KeywordsInput
      label="Validated Keywords"
      value={keywords}
      onChange={setKeywords}
      maxKeywords={5}
      minKeywordLength={3}
      maxKeywordLength={20}
      allowDuplicates={false}
      placeholder="Add 3-20 character keywords..."
      helperText={`${keywords.length}/5 keywords (3-20 chars, no duplicates)`}
    />
  );
};
```

### **Formik Integration**
```tsx
import React from 'react';
import { Formik, Form } from 'formik';
import * as Yup from 'yup';
import { FormikKeywordsInput } from '@app-studio/web';

const validationSchema = Yup.object({
  skills: Yup.array()
    .of(Yup.string().min(2, 'Skill must be at least 2 characters'))
    .min(1, 'At least one skill is required')
    .max(10, 'Maximum 10 skills allowed'),
});

export const FormikExample = () => {
  return (
    <Formik
      initialValues={{ skills: ['React', 'TypeScript'] }}
      validationSchema={validationSchema}
      onSubmit={(values) => console.log(values)}
    >
      <Form>
        <FormikKeywordsInput
          name="skills"
          label="Technical Skills"
          placeholder="Add your skills..."
          maxKeywords={10}
          helperText="Add your technical skills"
        />
      </Form>
    </Formik>
  );
};
```

## **Props**

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `value` | `string[]` | `undefined` | Controlled array of keywords |
| `defaultValue` | `string[]` | `[]` | Default keywords for uncontrolled component |
| `onChange` | `(keywords: string[]) => void` | `undefined` | Callback when keywords change |
| `onAdd` | `(keyword: string) => void` | `undefined` | Callback when keyword is added |
| `onRemove` | `(keyword: string, index: number) => void` | `undefined` | Callback when keyword is removed |
| `maxKeywords` | `number` | `undefined` | Maximum number of keywords allowed |
| `minKeywordLength` | `number` | `1` | Minimum length for each keyword |
| `maxKeywordLength` | `number` | `50` | Maximum length for each keyword |
| `allowDuplicates` | `boolean` | `false` | Whether to allow duplicate keywords |
| `separators` | `KeywordSeparator[]` | `['enter', 'comma']` | Keys that trigger keyword creation |
| `isRemovable` | `boolean` | `true` | Whether keywords can be removed |
| `placeholder` | `string` | `undefined` | Placeholder text for input |
| `label` | `string` | `undefined` | Label for the input field |
| `helperText` | `string` | `undefined` | Helper text below the input |
| `error` | `string \| boolean` | `false` | Error state or message |
| `size` | `'xs' \| 'sm' \| 'md' \| 'lg' \| 'xl'` | `'md'` | Size of the component |
| `variant` | `'outline' \| 'default' \| 'none'` | `'default'` | Visual variant |
| `shape` | `'default' \| 'sharp' \| 'rounded' \| 'pillShaped'` | `'default'` | Border radius style |
| `isDisabled` | `boolean` | `false` | Whether the input is disabled |
| `isReadOnly` | `boolean` | `false` | Whether the input is read-only |
| `views` | `KeywordsInputStyles` | `{}` | Custom styling object |

## **KeywordSeparator Types**
- `'enter'` - Enter key
- `'comma'` - Comma character
- `'space'` - Space character  
- `'tab'` - Tab key
