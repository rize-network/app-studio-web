# Gradient

Renders a customizable gradient background with various styles and animation options.

### **Import**
```tsx
import { Gradient } from '@app-studio/web';
```

### **Default**
```tsx
import React from 'react';
import { Gradient } from '@app-studio/web';

export const DefaultDemo = () => (
  <Gradient
    from="color.blue.500"
    to="color.purple.500"
    height="100px"
    width="100%"
  />
);
```
